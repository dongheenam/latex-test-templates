\examsection*{Expressions}

\begin{questions}
\question* Simplify the following as much as possible. 

\begin{mcparts}[2]
    \part $ 2a - 4b + 7b - 2a$ \points{2}
    \part $ 5x + x^2 + 2x^2 + 7x - 10$ \points{2}
    \vspace{2cm}
\end{mcparts}

\begin{answers}
    \begin{parts}
        \part $3b$
        \part $3x^2+12x-10$
    \end{parts}
\end{answers}

\question* Simplify the following expressions as much as possible.

\begin{mcparts}[2]
    \part $4x \times 4y$ $\vphantom{\dfrac{24abc}{3bd}}$ \points{2}
    \part $\dfrac{24abc}{3bd}$ \points{2}
    \vspace{2cm}
\end{mcparts}

\begin{answers}
    \begin{parts}
        \part $16xy$
        \part $\dfrac{8a}{d}$
    \end{parts}
\end{answers}

\question* Expand:

\begin{mcparts}[2]
    \part $-3(10u+v)$ \points{2}
    \part $2p(5p+7)$ \points{2}
    \vspace{2cm}
\end{mcparts}

\begin{answers}
    \begin{parts}
        \part $-30u-3v$
        \part $10p^2+14p$
    \end{parts}
\end{answers}

\question Factorise $21hm - 9mx$ as much as you can. \points{3}
\vspace{3cm}

\end{questions}

\emph{Solve one of the two questions below.}
\begin{questions}

\question Find two terms that add to $8x^2$. \marginpar{\llap{[2]}}
\begin{answers}
    $4x^2$, $4x^2$ (or other valid answers)

\question Expand and simplify $2x(x-1) - 3(x-7)$. \points{3}
\begin{answers}
    $2x^2 -5x + 21$
\end{answers}

\end{questions}

\examsection*{Index Laws}

\begin{questions}

\question Which one of following is the largest? \points{1}

\begin{choices*}
    \choice $10^2$
    \choice $2^9$
    \choice $7^3$
    \choice $3^6$
\end{choices*}
\vspace{1cm} 
\begin{answers}
    B
\end{answers}

\question* Simplify the following using index laws. You don't need to find the values.

\begin{mcparts}[2]
    \part $7^3 \times 7 \times 7^8$ $\vphantom{\dfrac{r^{12}}{r^3}}$ \points{1}
    \part $\dfrac{r^{12}}{r^3}$ \points{1}
    \vspace{1cm}
    \part $\left( 2b^2 \right)^3$ \points{2}
    \part $\left(n^3\right)^5 \div n^8$ \points{2}
    \vspace{2cm}
\end{mcparts}
\begin{answers}
    \begin{parts}
        \part $7^{12}$
        \part $r^9$
        \part $8b^6$
        \part $n^7$
    \end{parts}
\end{answers}
\end{questions}

\emph{Solve one of the three questions below.}

\begin{questions}
\question Fill in the blanks with numbers greater than 0 to make this equation true: \marginpar{\llap{[2]}}
$$ 2^{\boxed{\phantom{2^2}}} \times \left( 2^3 \right)^{\boxed{\phantom{2^2}}} = 2^{17} $$
\begin{answers}
$2$ and $5$, or other valid answers. 
\end{answers}

\question Evaluate: \marginpar{\llap{[3]}}
$$ \frac{2^{2022} \times 2^{2023} }{\left( 2^{2022} \right)^2 } $$ 
\begin{answers}
$2$
\end{answers}

\question Which power of $27$ is equal to $9^{300}$? \points{3}
\begin{answers}
$9^{300}=27^{200}$. 
\end{answers}

\end{questions}
